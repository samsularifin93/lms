<?php 

include 'database_connection.php';

include 'function.php';

if(!is_admin_login())
{
	header('location:../admin_login.php');
}

// Database connection info 
$dbDetails = $dbConfigArray;
 
// mysql db table to use 
$table = 'lms_author'; 
 
// Table's primary key 
$primaryKey = 'author_id'; 
 
// Array of database columns which should be read and sent back to DataTables. 
// The `db` parameter represents the column name in the database.  
// The `dt` parameter represents the DataTables column identifier. 
$columns = array( 
    array( 'db' => 'author_name', 'dt' => 0, 'field' => 'author_name' ), 
    array( 'db' => 'author_status',
			'dt' => 1,
			'field' => 'author_status',
			'formatter' => function( $d, $row ) {
				if($row["author_status"] == "Enable"){
					return '<span class="badge bg-success">'.$row["author_status"].'</span>';
				}else{
					return '<span class="badge bg-secondary">'.$row["author_status"].'</span>';
				}
			}
		), 
    array( 
        'db'        => 'author_created_on', 
        'dt'        => 2,
		'field' => 'author_created_on',
        'formatter' => function( $d, $row ) { 
            return date( 'jS M Y', strtotime($row['author_created_on'])); 
        } 
    ),
	array( 
        'db'        => 'author_updated_on', 
        'dt'        => 3,
		'field' => 'author_updated_on',
        'formatter' => function( $d, $row ) { 
            return date( 'jS M Y', strtotime($row['author_updated_on'])); 
        } 
    ),
    array( 
        'db'        => 'author_id',
        'dt'        => 4,
		'field' => 'author_id',
        'formatter' => function( $d, $row ) { 
            return '<a href="author.php?action=edit&code='.convert_data($row["author_id"]).'" class="btn btn-sm btn-primary">Edit</a><button name="delete_button" class="btn btn-danger btn-sm" onclick="delete_data('.$row["author_id"].')">Delete</button>'; 
        } 
    ) 
); 
 
// Include SQL query processing class 
require 'ssp.class.php'; 
 
// Output data as json format 
echo json_encode( 
    SSP::simple( $_GET, $dbDetails, $table, $primaryKey, $columns ));
